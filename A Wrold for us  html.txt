    <script>
        // --- 1. SNOWFALL ENGINE ---
        const canvas = document.getElementById('snow-canvas');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        const particles = [];
        
        for(let i=0; i<50; i++) particles.push({x: Math.random()*canvas.width, y: Math.random()*canvas.height, r: Math.random()*3+1, s: Math.random()*2+1});
        
        function drawSnow() {
            ctx.clearRect(0,0,canvas.width,canvas.height);
            ctx.fillStyle = "rgba(255, 255, 255, 0.8)";
            ctx.beginPath();
            particles.forEach(p => {
                ctx.moveTo(p.x, p.y);
                ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
                p.y += p.s;
                if(p.y > canvas.height) p.y = 0;
            });
            ctx.fill();
            requestAnimationFrame(drawSnow);
        }
        drawSnow();

        // --- 2. MUSIC PLAYER ---
        const audio = document.getElementById('bg-music');
        const vinyl = document.getElementById('vinyl');
        const status = document.getElementById('music-status');
        
        function toggleMusic() {
            if (audio.paused) {
                audio.play();
                vinyl.classList.add('spinning');
                status.innerText = "ON";
            } else {
                audio.pause();
                vinyl.classList.remove('spinning');
                status.innerText = "OFF";
            }
        }

        // --- 3. KATIHAR PARALLAX ---
        window.addEventListener('scroll', () => {
            const scroll = window.scrollY;
            const walkers = document.getElementById('walkers');
            const bubble = document.getElementById('walker-bubble');
            
            // Move walkers
            walkers.style.transform = `translateX(-50%) translateX(${scroll * 0.2}px)`;
            
            // Show bubble at specific scroll point
            if(scroll > 400 && scroll < 800) {
                bubble.style.opacity = 1;
            } else {
                bubble.style.opacity = 0;
            }
        });

        // --- 4. PCM HEART GRAPH ---
        const hCanvas = document.getElementById('heart-canvas');
        const hCtx = hCanvas.getContext('2d');
        hCtx.translate(150, 150);
        hCtx.scale(1, -1); 
        hCtx.strokeStyle = "#ff00ff";
        hCtx.lineWidth = 2;

        function drawHeartMath() {
            hCtx.beginPath();
            for (let t = 0; t <= Math.PI * 2; t += 0.05) {
                let x = 16 * Math.pow(Math.sin(t), 3);
                let y = 13 * Math.cos(t) - 5 * Math.cos(2 * t) - 2 * Math.cos(3 * t) - Math.cos(4 * t);
                hCtx.lineTo(x * 5, y * 5);
            }
            hCtx.stroke();
        }
        
        // Trigger drawing when visible
        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if(entry.isIntersecting) drawHeartMath();
            });
        });
        observer.observe(document.querySelector('.world-4'));

        // --- 5. CATCH GAME ---
        const gameArea = document.getElementById('game-area');
        const bucket = document.getElementById('player-bucket');
        const scoreEl = document.getElementById('score');
        let score = 0;

        gameArea.addEventListener('mousemove', (e) => {
            let rect = gameArea.getBoundingClientRect();
            let x = e.clientX - rect.left;
            if(x < 0) x = 0;
            if(x > 280) x = 280;
            bucket.style.left = x + 'px';
        });

        function spawnHeart() {
            let heart = document.createElement('div');
            heart.classList.add('falling-item');
            heart.innerText = '‚ù§Ô∏è';
            heart.style.left = Math.random() * 280 + 'px';
            heart.style.top = '0px';
            gameArea.appendChild(heart);

            let interval = setInterval(() => {
                let top = parseInt(heart.style.top);
                heart.style.top = (top + 3) + 'px';

                // Check collision
                let bRect = bucket.getBoundingClientRect();
                let hRect = heart.getBoundingClientRect();

                if(hRect.bottom >= bRect.top && hRect.right >= bRect.left && hRect.left <= bRect.right) {
                    score++;
                    scoreEl.innerText = score;
                    heart.remove();
                    clearInterval(interval);
                }

                if(top > 380) {
                    heart.remove();
                    clearInterval(interval);
                }
            }, 20);
        }
